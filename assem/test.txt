// crappy MIPS format
start:	lui   $r1, 0x20
	li    $r1, 0x00
	read  $r2, $r1, 0x00
	addi  $r2, 0x01
	write $r2, $r1, 0x00
	addi  $r1, 0x01
	jmp   start

// more readable?
start:	$r1.upper = 0x20 // lui
		$r1.lower = 0x00 // li
		M[$r1] => $r2    // read
		$r2 = $r2 + 0x01 // addi
		M[$r1] <= $r2    // write
		$r1 = $r1 + 0x01 // addi
		jump(start)      // jmp
